    JavaScript

    input-button.js js 

    import { UIElement, asBoolean, setText, setProperty } from '@efflore/ui-element'

class InputButton extends UIElement {
	static observedAttributes = ['disabled']
	static attributeMap = {
		disabled: asBoolean
	}

	connectedCallback() {
		this.first('button')
			.map(setText('label'))
			.forEach(setProperty('disabled'))
	}
}
InputButton.define('input-button')
